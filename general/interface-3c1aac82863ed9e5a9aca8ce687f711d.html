data HList :: [Type] -> Type where
  HN   :: HList '[]
  (:#) :: t -> HList ts -> HList (t:ts)

type family Fun (ts :: [Type]) (r :: Type) where
  Fun '[]    r = r
  Fun (t:ts) r = t -> Fun ts r

($#) :: Fun ts r -> HList ts -> r
f $# HN        = f
f $# (a :# as) = f a $# as
