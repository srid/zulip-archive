You can try yes, and find that the LHS is actually lazier than id, and instead a more precise RHS is \f x y -> f x y

{-# LANGUAGE TemplateHaskell #-}
module U where

import Test.Inspection

cu, id1, id2 :: (a -> b -> c) -> (a -> b -> c)
cu = curry . uncurry
id1 f = f
id2 f x y = f x y

inspect $ 'cu =/= 'id1
inspect $ 'cu ==- 'id2
